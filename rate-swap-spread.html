    if begin > leftmost:

    try:
        recs = fred_reader('USREC', start = datetime.datetime(1920, 1, 1)).query('USREC==1')
        recs_90 = recs.loc['1990':'1991']
        recs90_bgn = recs_90.index[0]

def fred_swap(old, ice, column_name):
    old_end = datetime.datetime(2014, 8, 1)
    swap_old = fred_reader(old)[:old_end]
    swap_old.columns = [column_name]
    swap_ice = fred_reader(ice)
    swap_ice.columns=[column_name]
    swap = pd.concat([swap_old, swap_ice]).drop_duplicates()
    return swap

swap_2y_ = fred_swap('WSWP2', 'ICERATES1100USD2Y', 'swap_2y')
swap_3y_ = fred_swap('WSWP3', 'ICERATES1100USD3Y', 'swap_3y')
swap_5y_ = fred_swap('WSWP5', 'ICERATES1100USD5Y', 'swap_5y')
ffr = fred_reader('DFF')

rates = ffr.join([swap_2y_, swap_3y_, swap_5y_]).dropna()
