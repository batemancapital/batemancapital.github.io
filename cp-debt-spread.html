expire_after = datetime.timedelta(days=1)
session = requests_cache.CachedSession(cache_name='cache', backend='sqlite', expire_after=expire_after)

start = datetime.datetime(1990, 6, 1)
end = datetime.date.today()

def fred_reader(symbol):
    return pd_reader.data.DataReader(symbol, 'fred', start, end, session=session)

def try_draw_span(ax, begin, end, leftmost):
    if begin > leftmost:
        ax.axvspan(begin, end, color='tab:grey', alpha=0.3)

def plot_recession(ax, leftmost):
    recs = fred_reader('USREC').query('USREC==1')
    recs_90 = recs.loc['1990':'1991']
    recs_2k = recs.loc['2001']
    recs_2k8 = recs.loc['2008':]
    recs90_bgn = recs_90.index[0]
    recs90_end = recs_90.index[-1]
    recs2k_bgn = recs_2k.index[0]
    recs2k_end = recs_2k.index[-1]
    recs2k8_bgn = recs_2k8.index[0]
    recs2k8_end = recs_2k8.index[-1]
    try_draw_span(ax, recs90_bgn, recs90_end, leftmost)
    try_draw_span(ax, recs2k_bgn, recs2k_end, leftmost)
    try_draw_span(ax, recs2k8_bgn, recs2k8_end, leftmost)
    
def add_source(ax, annotation):
    plt.text(0, -0.1, annotation,
      horizontalalignment='left',
      verticalalignment='center',
      fontsize=14,
      transform = ax.transAxes)
    plt.text(0, -0.15, ": Bateman Capital",
      horizontalalignment='left',
      verticalalignment='center',
      fontsize=14,
      transform = ax.transAxes)
