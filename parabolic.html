start = datetime.datetime(2017, 9, 1)
tsla = yahoo_reader('TSLA', start)
btc = yahoo_reader('BTC-USD').shift(periods=985, freq='D')

def draw(start, end=datetime.datetime(2023, 1, 1)):
    fig, ax = plt.subplots()
    ax2 = ax.twinx()
    ax.set_ylabel('$/share')
    ax.plot(tsla[start:].Close, color='green', label='TSLA')
    ax2.plot(btc[start:end].Close, color='red', label='BTC', alpha=0.3)
    ax.legend(loc='upper left')
    ax2.legend(loc='lower right')
    add_source(ax, "Yahoo Finance")
    plt.title('TSLA vs BTC(+985days)')
    
draw(start)
